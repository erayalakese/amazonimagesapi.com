Amazon Images API

Fetch images of any ASIN via Amazon Selling Partner API.

var requestOptions = {
  method: 'GET'
};

fetch("https://api.amazonimagesapi.com/:ASIN:/:domain:",
requestOptions)
  .then(response => response.text())
  .then(result => console.log(result))
  .catch(error => console.log('error', error));
